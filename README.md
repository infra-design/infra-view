```jsx
  display='block'
  margin={100}
  width={20}
  height={20}
  background='black'
  hover={{
    background='red'
  }}
  md={{
    margin: 200,
    width: 30,
    height: 30,
  }}
  lg={{
    margin: 300,
    width: 30,
    height: 30,
  }}>
  UI
</View>
```
